130  constexpr bool is_prime(T x) {
131  auto n = static_cast<std::uint64_t>(x);
132  if (n <= 1)
133  return 0;
134  else if (n <= 3)
135  return 1;
136  else if (!(n & 1))
137  return 0;
138  else if (!(n % 3))
139  return 0;
140  std::uint64_t r = 5;
141  while (r * r <= n) {
142  if ((n % r) == 0 || (n % (r + 2)) == 0)
143  return 0;
144  r += 6;
145  }
146  return false;
147  }
